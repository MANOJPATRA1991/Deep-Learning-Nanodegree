3. View an Image in More Detail¶

In [10]:
# Any grayscale image is interpreted by the computer as a matrix
# with one entry for each image pixel

# Each image in the MNIST database is 28 pixels high and 28 pixels wide and
# so, understood by the computer as a 28 by 28 matrix 
# => 784 (28 X 28) entries as a vector => 784 input layers
# and 10 output layers

# Whit pixels encoded as 255, black pixels as 0 and gray pixels as 114

# Multi-layer perceptron (MLP) take vectors as inputs to input layers. So,
# we need to convert the images into vectors

def visualize_input(img, ax):
    ax.imshow(img, cmap='gray')
    width, height = img.shape
    thresh = img.max()/2.5
    for x in range(width):
        for y in range(height):
            ax.annotate(str(round(img[x][y],2)), xy=(y,x),
                        horizontalalignment='center',
                        verticalalignment='center',
                        color='white' if img[x][y]<thresh else 'black')

fig = plt.figure(figsize = (12,12)) 
ax = fig.add_subplot(111)
visualize_input(X_train[0], ax)

6. Define the Model Architecture¶

In [20]:
from keras.models import Sequential
from keras.layers import Dense, Dropout, Flatten

# define the model
model = Sequential()
# Flatten layer is used to flatten the input and appears before
# we specify the MLP. It takes the image matrix's input and 
# converts it into a vector
model.add(Flatten(input_shape=X_train.shape[1:]))

# 'relu' activation function leaves all positive values alone and
# sends all negative values to zero
model.add(Dense(512, activation='relu'))

# Dropout consists in randomly setting a fraction rate of 
# input units to 0 at each update during training time, 
# which helps prevent overfitting.
model.add(Dropout(0.2))

model.add(Dense(512, activation='relu'))

# Dropout consists in randomly setting a fraction rate of 
# input units to 0 at each update during training time, 
# which helps prevent overfitting.
model.add(Dropout(0.2))

# Output layer with 10 nodes, where each node 'i' depicts
# the probability that the image depicted is 'i'
model.add(Dense(10, activation='softmax'))

# summarize the model
model.summary()

# Fit the model weights based on the TRAIN SET
# Check how the model is doing based on the VALIDATION SET
# Check the accuracy of the trained model based on the TEST SET

# train the model
# ModelCheckpoint class saves the model after each epoch
# Arguments: filepath --> where to save the model weights
#            verbose --> 0, 1 or 2
#            save_best_only --> If true, we ask to save only the weights to 
#                               to get the best accuracy on the validation set

                               verbose=1, save_best_only=True)
hist = model.fit(X_train, y_train, batch_size=128, epochs=10,
          validation_split=0.2, callbacks=[checkpointer],
          verbose=1, shuffle=True)
